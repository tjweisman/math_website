 54class CP1Disk(CP1Object):
 55    def _compute_proj_data(self, center, rad, radius_metric="affine"):
 56        # given the center/radius of a disk in affine coordinates,
 57        # compute projective coordinates of three points on the boundary of the disk
 58        # and the center.
 59        if radius_metric == "affine":
 60            center_coords = center.real_affine_coords()
 61
 62            with np.errstate(divide="ignore", invalid="ignore"):
 63                normed_ctr = utils.normalize(center_coords)
 64                # we can just pick something arbitrary (unit-length)
 65                # if we're at the origin
 66                normed_ctr[utils.normsq(center_coords) == 0] = np.array([1.0, 0.0])
 67
 68            perp_ctr = np.stack([-1 * normed_ctr[..., 1], normed_ctr[..., 0]],
 69                                axis=-1)
 70
 71            e_rad = np.expand_dims(rad, axis=-1)
 72
 73            affine_real_pts = np.stack([center_coords + e_rad * normed_ctr,
 74                                        center_coords - e_rad * normed_ctr,
 75                                        center_coords + e_rad * perp_ctr,
 76                                        center_coords], axis=-2)
 77
 78            affine_c_pts = utils.r_to_c(affine_real_pts)
 79            return np.stack([np.ones_like(affine_c_pts), affine_c_pts],
 80                            axis=-1)
 81        elif radius_metric == "fs":
 82            center_coords = center.spherical_coords()
 83
 84            q, r = np.linalg.qr(np.expand_dims(center_coords, axis=-1),
 85                                mode='complete')
 86
 87            p1_t = np.stack([np.cos(2 * rad),
 88                             np.sin(2 * rad),
 89                             np.zeros_like(rad)], axis=-1)
 90            p2_t = np.stack([np.cos(2 * rad),
 91                             -np.sin(2 * rad),
 92                             np.zeros_like(rad)], axis=-1)
 93            p3_t = np.stack([np.cos(2 * rad),
 94                             np.zeros_like(rad),
 95                             np.sin(2 * rad)], axis=-1)
 96
 97            t_pts = np.stack([p1_t, p2_t, p3_t], axis=-1)
 98            sph_pts = (np.expand_dims(r[..., 0, 0], axis=(-1, -2)) *
 99                       q @ t_pts)
100
101            proj_pts = spherical_to_projective(
102                sph_pts.swapaxes(-1, -2)
103            )
104
105            return np.concatenate([proj_pts,
106                                   np.expand_dims(center.projective_coords(),
107                                                  axis=-2)], axis=-2)

128    def circle_parameters(self):
129        bdry_pts = self.boundary_points()
130        bdry_aff_coords = bdry_pts.real_affine_coords()
131
132        # there's a more pythonic way to do this but IDGAF

138    def center_inside(self):
139        circ_ctr, circ_rad = self.circle_parameters()
140        ipts = self.interior_point()
141
142        center_affine = ipts.in_affine_chart(0)
143        int_pt_coords = ipts[center_affine].real_affine_coords()
144        dist_sq = utils.normsq(circ_ctr[center_affine] - int_pt_coords)
145
146        res = np.full(self.shape, False)
147
148        res[center_affine] = dist_sq < circ_rad[center_affine]**2
149
150        return res

152    def fs_diameter(self):
153        center, radius = self.circle_parameters()
154        center_norm = np.linalg.norm(center, axis=-1)
155
156        res = np.arctan(center_norm + radius) - np.arctan(center_norm - radius)
157        inverted = ~self.center_inside()
158        res[inverted] = np.pi - res[inverted]
159
160        return res
161
162    def fs_center(self):
163        center, radius = self.circle_parameters()
164        center_norm = np.linalg.norm(center, axis=-1)
165
166        fs_ctr_dist = np.tan(
167            (np.arctan(center_norm + radius) +
168             np.arctan(center_norm - radius)) / 2
169        )
170
171        scale_factor = np.ones_like(center_norm)
172        nonzero_scale = np.abs(fs_ctr_dist) > 0
173        scale_factor[nonzero_scale] = (fs_ctr_dist[nonzero_scale] /
174                                       center_norm[nonzero_scale])
175
176        fs_center = CP1Point(np.expand_dims(scale_factor, axis=-1) * center,
177                             coords="real_affine")
178
179        # apply a sphere inversion for disks containing infinity
180        inverted = ~self.center_inside()
181        inverted_pts = inversion() @ fs_center[inverted]
182        fs_center[inverted] = inverted_pts
183
184        fs_center.proj_data[inverted] = np.conjugate(
185            fs_center.proj_data[inverted]
186        )
187
188        return fs_center

190    def inversion(self):
191        basechange = self.boundary_points().to_standard_triple()
192        std_inversion = projective.Transformation(
193            np.array([[1.0, 0.], [0., -1.]])
194        )
195        return basechange.inv() @ std_inversion @ basechange

197    def complement(self):
198        new_int = self.inversion() @ self.interior_point()

200        complement = CP1Disk(copy(self.proj_data))
201        complement.proj_data[..., -1, :] = new_int.proj_data
202        return complement

204    def contains(self, other, broadcast="elementwise"):
205        if broadcast == "pairwise":
206            s_to_use = self.flatten_to_unit()
207            o_to_use = other.flatten_to_unit()
208        else:
209            s_to_use = self
210            o_to_use = other
211
212        s_aff = s_to_use.center_inside()
213        o_aff = o_to_use.center_inside()
214
215        sctr, srad = s_to_use.circle_parameters()
216        octr, orad = o_to_use.circle_parameters()
217
218        contain, contained, intersect = utils.disk_interactions(
219            sctr, srad, octr, orad, broadcast=broadcast
220        )
221
222        res = np.full(contain.shape, False)
223
224        if broadcast == "elementwise":
225            res[s_aff & o_aff] = contain[s_aff & o_aff]
226            res[~s_aff & o_aff] = ~intersect[~s_aff & o_aff]
227            res[~s_aff & ~o_aff] = contained[~s_aff & ~o_aff]
228            return res
229
230        affaffmask = np.logical_and(
231            np.expand_dims(s_aff, axis=1),
232            np.expand_dims(o_aff, axis=0)
233        )
234        naffaffmask = np.logical_and(
235            np.expand_dims(~s_aff, axis=1),
236            np.expand_dims(o_aff, axis=0)
237        )
238        naffnaffmask = np.logical_and(
239            np.expand_dims(~s_aff, axis=1),
240            np.expand_dims(~o_aff, axis=0)
241        )
242        np.putmask(res, affaffmask, contain)
243        np.putmask(res, naffaffmask, ~intersect)
244        np.putmask(res, naffnaffmask, contained)
245        return res

290def projective_to_spherical(points, column_vectors=False):
291    ppoints = np.array(points)
292    if column_vectors:
293        ppoints = spoints.swapaxes(-1, -2)
294
295    z0 = ppoints[..., 0]
296    z1 = ppoints[..., 1]
297    normsq = np.abs(z0 * np.conjugate(z0) + z1 * np.conjugate(z1))
298    horizontal = utils.c_to_r(2 * np.conjugate(z0) * z1 / normsq)
299    vertical = (np.abs(z1 * np.conjugate(z1)) -
300                np.abs(z0 * np.conjugate(z0))) / normsq
301
302    spherical = np.concatenate([horizontal, np.expand_dims(vertical, axis=-1)],
303                               axis=-1)
304
305    if column_vectors:
306        spherical = spherical.swapaxes(-1, -2)
307
308    return spherical

310def spherical_to_projective(points, column_vectors=False):
311    spoints = np.array(points)
312    if column_vectors:
313        spoints = spoints.swapaxes(-1, -2)
314
315    chart2 = spoints[..., -1] > 0
316
317    res = np.stack([1 - spoints[..., -1],
318                    utils.r_to_c(spoints[..., :-1])],
319                   axis=-1)
320
321    # need to do this in two charts
322    res[chart2] = np.stack(
323        [np.conjugate(utils.r_to_c(spoints[chart2, :-1])),
324         1 + spoints[chart2, -1]],
325        axis=-1)
326
327    if column_vectors:
328        res = res.swapaxes(-1, -2)
329
330    return res

332def to_standard_triple(triple):
333    tpoint = CP1Point(triple)
334    if tpoint.proj_data.shape[-2] != 3:
335        raise GeometryError(
336            f"Cannot take an array of {self.proj_data.shape[-2]}-tuples"
337            " to an array of triples."
338        )
339
340    tdata = tpoint.projective_coords()
341
342    res = utils.invert(tdata[..., :2, :])
343    p3_t = np.expand_dims(tdata[..., 2, :],
344                          axis=-2) @ res
345
346    eigenvalue = np.emath.sqrt(p3_t[..., 1] / p3_t[..., 0])
347
348    res[..., 0] *= eigenvalue
349    res[..., 1] /= eigenvalue
350
351    return projective.Transformation(res)

Represent some object in projective geometry (possibly a composite object).

The underlying data of a projective object is stored as a numpy ndarray. The last unit_ndims ndims of this array describe a single instance of this type of object.

For example, a Polygon object has unit_ndims equal to 2, since a single Polygon is represented by an array of shape (n,d), where n is the number of vertices and d is the dimension of the underlying vector space. So, a Polygon object whose underlying array has shape (5, 6, 4, 3) represents a 5x6 array of quadrilaterals in RP^2 (i.e. the projectivization of R^3).

Represent some object in projective geometry (possibly a composite object).

The underlying data of a projective object is stored as a numpy ndarray. The last unit_ndims ndims of this array describe a single instance of this type of object.

For example, a Polygon object has unit_ndims equal to 2, since a single Polygon is represented by an array of shape (n,d), where n is the number of vertices and d is the dimension of the underlying vector space. So, a Polygon object whose underlying array has shape (5, 6, 4, 3) represents a 5x6 array of quadrilaterals in RP^2 (i.e. the projectivization of R^3).

Parameters
  • proj_data (ndarray): underyling data describing this projective object
  • aux_data (ndarray): auxiliary data describing this projective object. Auxiliary data is any data which is in principle computable from proj_data, but is convenient to keep as part of the object definition for transformation purposes.
  • dual_data (ndarray): data describing this projective object which transforms covariantly, i.e. as a dual vector in projective space.
  • unit_ndims (int): number of ndims of an array representing a "unit" version of this object. For example, an object representing a single point in hyperbolic space has unit_ndims 1, while an object representing a line segment has unit_ndims equal to 2.
  • aux_ndims (int): like unit_ndims, but for auxiliary data.
  • dual_ndims (int): like unit_ndims, but for covariant (dual) data.
  • base_ring (object): (requires sage). Base ring for matrix entries of the underlying data for this object. This allows for exact computations in projective space (at the cost of performance) by supplying a ring that supports exact computations. The dtype of the underlying data is automatically converted to "object" if this is specified.
  • rational_approx (bool): (requires sage). If base_ring is specified, coerce floating-point data to a rational approximation before converting to the specified ring. This is useful for combining computations with floating-point values with computations in algebraic number fields.
