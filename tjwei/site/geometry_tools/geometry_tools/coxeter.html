 20    def __init__(self, diagram=None, matrix=None, generator_style="alpha"):
 21        """Construct a Coxeter group.
 22
 23        The constructor accepts data determining a Coxeter group in
 24        the form of either a Coxeter diagram (specified as an iterable
 25        of tuples) or a Coxeter matrix (a symmetric array of
 26        integers).
 27
 28        Parameters
 29        ----------
 30        diagram: iterable of tuples
 31            A Coxeter diagram, given as an iterable of triples of the
 32            form `(generator1, generator2, order)`. Each generator can
 33            be any hashable object. `order` is an integer. If it is
 34            positive, then it represents the order of the product of
 35            generator1 and generator2. If it is zero or negative, the
 36            order of this product is infinite.
 37
 38        matrix: ndarray of ints
 39            The Coxeter matrix defining this Coxeter system. This must
 40            be a symmetric matrix of integers, with 1's on the
 41            diagonal.
 42
 43        generator_style : "alpha" or "alphanum"
 44            Default naming convention to use for Coxeter group
 45            generators. If "alpha", then generators are named "a",
 46            "b", "c", etc. If "alphanum", then generators are named
 47            "s0", "s1", "s2", etc.
 48
 49            This parameter is ignored if a diagram is specified
 50            (generator names will be taken from the diagram keys
 51            instead).
 52
 53        """
 54        if diagram is None and matrix is None:
 55            raise ValueError("Must provide data to construct a Coxeter group")
 56
 57        if diagram is not None and matrix is not None:
 58            warnings.warn("Redundant Coxeter data specified; ignoring Coxeter matrix and"
 59                          " constructing from diagram.")
 60
 61        if diagram is not None:
 62            self.from_diagram(diagram)
 63        elif matrix is not None:
 64            self.from_coxeter_matrix(matrix, generator_style)
 65
 66    def bilinear_form(self, **kwargs):
 67        base_ring, dtype = utils.check_type(**kwargs)
 68        pi = utils.pi(**kwargs)
 69        half = utils.number(0.5, like=pi)
 70
 71        adjusted_cox_matrix = utils.array_like(
 72            self.coxeter_matrix, like=half
 73        )
 74        adjusted_cox_matrix[adjusted_cox_matrix.astype(float) <= 0] = half
 75
 76        result = utils.change_base_ring(
 77            -1 * np.cos(pi / adjusted_cox_matrix),
 78            base_ring=base_ring
 79        )
 80        if dtype is None:
 81            return result
 82
 83        return result.astype(dtype)

 85    def from_diagram(self, diagram):
 86        self.generators = defaultdict(dict)
 87
 88        for edge in diagram:
 89            g1, g2, order = edge
 90            self.generators[g1][g2] = order
 91            self.generators[g2][g1] = order
 92
 93        self.generator_index = {g:i for i, g in enumerate(self.generators)}
 94        self.ordered_gens = [None] * len(self.generators)
 95        for g, i in self.generator_index.items():
 96            self.ordered_gens[i] = g
 97            self.generators[g][g] = 1
 98
 99        self.coxeter_matrix = np.array([
100            [self.generators[g1][g2] for g2 in self.ordered_gens]
101            for g1 in self.ordered_gens
102        ])

106            self.__class__.__name__, self.coxeter_matrix

122    def from_coxeter_matrix(self, matrix, generator_style="alpha"):
123        _matrix = np.array(matrix)
124        if (_matrix.T - _matrix != 0).any():
125            raise GeometryError("Coxeter matrix must be symmetric")
126
127        if (_matrix.astype(int) != _matrix).any():
128            raise GeometryError("Coxeter matrix must have integer entries")
129
130        if (np.diag(_matrix) != 1).any():
131            warnings.warn("Coxeter matrix should have 1's on the diagonal")
132
133        self.coxeter_matrix = _matrix
134        self.generators = defaultdict(dict)
135
136        n = len(_matrix)
137        self.generator_index = {
138            CoxeterGroup._default_generator_name(i, generator_style):i
139            for i in range(n)
140        }
141
142        self.ordered_gens = [
143            CoxeterGroup._default_generator_name(i, generator_style)
144            for i in range(n)
145        ]
146
147        for i, row in enumerate(_matrix):
148            for j, order in enumerate(row):
149                self.generators[self.ordered_gens[i]][self.ordered_gens[j]] = order

151    def cartan_representation(self, cartan_matrix,
152                              rename_generators=False,
153                              generator_style="alpha",
154                              diagonalize=False,
155                              order_eigenvalues="signed",
156                              **kwargs):
157        """Find a representation of this Coxeter group determined by a Cartan
158        matrix.
159
160        Parameters
161        ----------
162        cartan_matrix : ndarray
163            The n*n matrix determining a representation of this
164            Coxeter group into PGL(n, R), generated by reflections.
165        rename_generators : bool
166            If True, then rename the generators according to the style
167            in `generator_style`. If False (the default), use the
168            provided names for the generators.
169        generator_style : 'alpha' or 'alphanum'
170            Which convention to use when renaming generators if
171            rename_generators is True.
172        diagonalize : bool
173            If True, and the specified Cartan matrix is symmetric,
174            conjugate the computed representation so that it preserves
175            a diagonal symmetric bilinear form with unit-modulus
176            eigenvalues.
177        order_eigenvalues: {'signed', 'minkowski'}
178            If diagonalize is True, specify how to order the diagonal
179            basis for the form. See the documentation for
180            utils.diagonalize_form for details.
181
182        Returns
183        -------
184        representation.Representation
185            Representation determined by the Cartan matrix
186
187        Raises
188        ------
189        ValueError
190            Raised if rename_generators is False and any generator has
191            a name which is not a single character.
192
193    """
194        num_gens = len(self.generators)
195        rep = Representation()
196
197        base_ring, dtype = utils.check_type(**kwargs)
198
199        for i, gen in enumerate(self.ordered_gens):
200            basis_vec = utils.zeros(num_gens, like=cartan_matrix)
201            basis_vec[i] = utils.number(1, like=cartan_matrix)
202            diagonal = np.diag(basis_vec)
203
204            g_name = gen
205            if rename_generators:
206                g_name = CoxeterGroup._default_generator_name(i, generator_style)
207
208            gen_val = (utils.identity(num_gens, like=cartan_matrix) -
209                       diagonal @ cartan_matrix)
210
211            gen_val = utils.change_base_ring(gen_val, base_ring)
212
213            if dtype is not None:
214                gen_val = gen_val.astype(dtype)
215
216            rep[g_name] = gen_val
217
218        if not diagonalize:
219            return rep
220
221        #TODO: check if Cartan matrix is symmetric for this step
222        W, Winv = utils.diagonalize_form(
223            cartan_matrix / 2,
224            order_eigenvalues=order_eigenvalues,
225            with_inverse=True
226        )
227
228        W = utils.change_base_ring(W, base_ring)
229        Winv = utils.change_base_ring(Winv, base_ring)
230
231        return rep.compose(
232            lambda mat: Winv @ mat @ W
233        )

235    def geometric_representation(self, rename_generators=False,
236                                 diagonalize=False,
237                                 order_eigenvalues="signed",
238                                 **kwargs):
239        """Get the geometric representation of the Coxeter group, i.e. the
240        dual of the canonical representation.
241
242        See the documentation for CoxeterGroup.cartan_representation
243        for descriptions of additional keyword arguments.
244
245        Returns
246        -------
247        representation.Representation
248            The geometric representation for this Coxeter group, which
249            acts discrete faithfully on a convex subset of projective
250            space RP^{n-1}.  This representation always preserves the
251            symmetric bilinear form determined by the cosine matrix
252            (which is in turn determined by the Coxeter matrix).
253
254        """
255        return self.cartan_representation(
256            2 * self.bilinear_form(**kwargs),
257            rename_generators=False,
258            diagonalize=diagonalize,
259            order_eigenvalues="signed",
260            **kwargs
261        )

263    def canonical_representation(self, **kwargs):
264        """Get the canonical representation of this Coxeter group into PGL(n,
265        R).
266
267        See the documentation for CoxeterGroup.cartan_representation
268        for descriptions of additional keyword arguments.
269
270        Returns
271        -------
272        representation.Representation
273            The canonical (Tits) representation for this Coxeter
274            group.
275
276        """
277        return self.geometric_representation(**kwargs).compose(
278            lambda mat: utils.invert(mat.T)

282    def cartan_matrix(self, parameters, **kwargs):
283        """Get a Cartan matrix for this Coxeter group by specifying
284        off-diagonal parameters.
285
286        Parameters
287        ----------
288        parameters : dict or ndarray
289            dictionary specifying free parameters for a Cartan matrix,
290            i.e. entries of the Cartan matrix corresponding to pairs
291            of generators s,t where the group element s*t has infinite
292            order.
293
294            Parameters are specified either as a dictionary or a
295            matrix. To specify parameters as a dictionary, use the
296            format
297
298            {(i,j):v, (k, l): u, ...}
299
300            where i,j,k,l etc. are generator indices, and v, u,
301            etc. are values of the Cartan matrix.
302
303            To specify parameters as a matrix, pass an n*n matrix with
304            all entries zero except for the free parameters in the
305            Cartan matrix.
306
307            For either format, if the (i,j) entry of the Cartan matrix
308            is set, but the (j,i) entry of the matrix is not set (or
309            set to zero), then this function assumes that the intended
310            Cartan matrix is symmetric and will set the (j,i) entry
311            equal to the (i,j) entry.
312
313        Returns
314        -------
315        ndarray
316            Cartan matrix, specified by the parameters as described above.
317
318        """
319        cartan = 2 * self.bilinear_form(**kwargs)
320
321        def specified(index):
322            try:
323                return parameters[index] != 0
324            except KeyError:
325                return False
326
327        for index in zip(*np.nonzero(self.coxeter_matrix < 0)):
328            if specified(index):
329                cartan[index] = parameters[index]
330                r_index = index[::-1]
331                if not specified(r_index):
332                    cartan[r_index] = parameters[index]
333
334        return cartan

360        """Get a representation of this Coxeter group in PO(d, 1).
361

384    def automaton(self, shortlex=True, even_length=False):
385        """Get a finite-state automaton accepting geodesic words in the
386        Coxeter group.
387
388        Parameters
389        ----------
390        shortlex : bool
391            If True (the default), return an automaton which only
392            accepts shortlex geodesic representatives of elements in
393            the group. If False, return an automaton which only
394            accepts geodesic words, and accepts at least one geodesic
395            word per element.
396        even_length : bool
397            If True, return an automaton only accepting words of even
398            length.
399
400        Returns
401        -------
402        fsa.FSA
403            finite-state automaton accepting geodesic words.
404
405        """
406        aut = coxeter_automaton.generate_automaton_coxeter_matrix(
407            self.coxeter_matrix,
408            shortlex
409        )
410        aut.rename_generators(self.ordered_gens)
411
412        if even_length:
413            return aut.even_automaton()
414
415        return aut

417    def standard_subgroup(self, generators):
418        """Get a standard subgroup of the Coxeter group, i.e. a subgroup
419        generated by a subset of the standard set.
420
421        Parameters
422        ----------
423        generators : iterable
424            generating set for the standard subgroup
425
426        Returns
427        -------
428        CoxeterGroup
429            Standard (Coxeter) subgroup with the given generating set
430        """
431
432        subdiagram = []
433        for g1, g2s in self.generators.items():
434            if g1 not in generators:
435                continue
436            for g2, order in g2s.items():
437                if g2 not in generators:
438                    continue
439                subdiagram.append((g1, g2, order))
440
441        return CoxeterGroup(diagram=subdiagram)
